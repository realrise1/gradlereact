# springboot , gradle , java17, mysql8, react
# mysql workbench8 설치
# react 설치
> cd src/main 해서
> 
> npm install -g create-react-app
> 
> cd frontend

> npm install axios --save
> 
> npm install react-router-dom --save
> 
> npm install http-proxy-middleware --save
> 
> npm start
